package Homework;

/**
 *
 * @author User
 */
public class FailedStd extends Subject {

    private String stdName = null;
    private double score = 0;

    public FailedStd() {
    }

    public FailedStd(String id, String name, int credit, String stdName, double score) {
        super(id, name, credit);
        this.stdName = stdName;
        this.score = score;

    }

    @Override
    public String toString() {
        return "FailedStd" + "\nstdName : " + stdName + "\nscore : " + score;
    }

}

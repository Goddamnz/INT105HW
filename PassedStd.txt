package Homework;

/**
 *
 * @author User
 */
public class PassedStd extends Subject {

    private String stdName = null;
    private double stdGrade = 1;
    private double score = 0;
    private boolean isMax = false;

    public PassedStd() {
    }

    public PassedStd(String id, String name, int credit, String stdName, double stdGrade, double score, boolean isMax) {
        super(id, name, credit);
        this.stdName = stdName;
        this.stdGrade = stdGrade;
        this.score = score;
        this.isMax = isMax;

    }

    public String getStdName() {
        return stdName;
    }

    public void setStdName(String stdName) {
        this.stdName = stdName;
    }

    public double getStdGrade() {
        return stdGrade;
    }

    public void setStdGrade(double stdGrade) {
        this.stdGrade = stdGrade;
    }

    public double getScore() {
        return score;
    }

    public void setScore(double score) {
        this.score = score;
    }

    public boolean isIsMax() {
        return isMax;
    }

    public void setIsMax(boolean isMax) {
        this.isMax = isMax;
    }

    public boolean rewardForMax() {
        boolean getReward = false;

        if (this.isMax == true) {
            getReward = true;
        }

        return getReward;
    }

    @Override
    public String toString() {
        return "PassedStd" + "\nstdName : " + stdName + "\nstdGrade : " + stdGrade + "\nscore : " + score + "\nisMax : " + isMax;
    }

}

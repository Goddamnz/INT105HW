        boolean reward=false;
        
        
        Subject subject1 = new Subject("INT102","Computer programming",3);
        System.out.println(subject1.toString());
        
        PassedStd std1 = new PassedStd("INT102","Computer programming",3,"Student a",3.5,80,true);
        System.out.println(std1.toString());

        }
        
        FailedStd std2 = new FailedStd("INT102","Computer programming",3,"Student b",30);
        System.out.println(std2.toString());
        
        
        
        
        
    }
